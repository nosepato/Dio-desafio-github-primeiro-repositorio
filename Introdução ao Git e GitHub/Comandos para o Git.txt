dir ou ls - listar as pastas
cd/ - navegar entre as pastas
cd.. - retroceder um passo
cls ou clear - limpar o terminal
botão TAB - auto completa

mkdir - criar uma pasta
echo - printar de volta no terminal oque for digitar
> - redericionar fluxo(criar o arquivo)
del - deletar arquivos
seta pra cima - volta os comandos usados
rmdir "pasta" /s/q - (não lembro)
ls-a - mostrar arquivos ocultos

comando para criar chave SSH: ssh-keygen -t ed2559 -c "e-mail"
comando para inicializar o SSH agent: eval $(ssh-agent-s)
comando para entregar a chave: ssh-add id...
autor para o commit: git config --global user.name...
adicionar commit: git add*
                  git commit -m "commit inicial"
comando parar monitorar os status dos arquivos: git status
comando para mover o arquivo: mv "nome do arquivo" ./ "nome da pasta"
comando para alterar as configurações do git: git config --global --unset user."email" ou "name"
comando para listar as configurações: git config --list
comando para enviar o repositório da maquina para o github remoto: git remote add origin "url"
comando para listar is repositórios remotos: git remote -v
comando para enviar o repositório: git push origin master
comando para puxar do github para a máquina: git pull origin master
comando para clonar o repositório do github para a máquina: git clone "url"
